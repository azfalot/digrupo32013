CREATE TABLE ESCALADORES (
P_ESCALADORES INTEGER PRIMARY KEY,
NOMBRE VARCHAR(30) ,
WALLPAPER VARCHAR(100),
PERIODO_INICIO DATE,
PERIODO_FIN DATE
);

CREATE TABLE ITINERARIO (
P_ITINERARIO INTEGER PRIMARY KEY,
NOMBRE VARCHAR(30),
DIFICULTAD VARCHAR(30),
LOCALIZACION VARCHAR(30),
TIPO INTEGER,
FOTO VARCHAR(100)
);

CREATE TABLE SESION_ENTRENAMIENTOS (
P_SESION_ENTRENAMIENTOS INTEGER PRIMARY KEY,
A_ESCALADORES INTEGER,
HORA_INICIO TIME,
HORA_FIN TIME,
FECHA DATE,
TIPO INTEGER,
DESCRIPCION VARCHAR(255)
);

CREATE TABLE ESC_IT (
P_ESC_IT INTEGER PRIMARY KEY,
A_ESCALADORES INTEGER,
A_ITINERARIO INTEGER,
FECHA DATE
);

ALTER TABLE PUBLIC.SESION_ENTRENAMIENTOS
ADD FOREIGN KEY(A_ESCALADORES) REFERENCES PUBLIC.ESCALADORES(P_ESCALADORES) ON DELETE CASCADE

ALTER TABLE PUBLIC.ESC_IT
ADD FOREIGN KEY(A_ESCALADORES) REFERENCES PUBLIC.ESCALADORES(P_ESCALADORES) ON DELETE CASCADE

ALTER TABLE PUBLIC.ESC_IT
ADD FOREIGN KEY(A_ITINERARIO) REFERENCES PUBLIC.ITINERARIO(P_ITINERARIO) ON DELETE CASCADE
